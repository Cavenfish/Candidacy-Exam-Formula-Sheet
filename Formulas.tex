     \Large{\textbf{Electrodynamics}} \\
     \hrule
\end{center}

\section*{Electricity}
Gauss' Law:
$\Phi_E = \oint \vec{E} \cdot \vec{dA} = \frac{Q_{enc}}{\epsilon_{\circ}}$

Electric Displacement:
$\vec{D} = \epsilon_{\circ} \vec{E} + \vec{P}$

Combine above two:
$\oint \vec{D} \cdot \vec{dA} = Q_{free}$

Free charge density:
$\rho_f =  \vec{\nabla} \cdot \vec{D}$

Bound charge density:
$\rho_b = -\vec{\nabla} \cdot \vec{P}$

Bound surface charge density:
$\sigma_b = \vec{P} \cdot \hat{n}$

Electric potential from electric field:
$V = \int E dr$

General solution for potential with r,$\theta$ dependence:
$V(r,\theta) \sum_{l=0}^{\infty} (A_lr^l +\frac{B_l}{r^{l+1}})P_l(\cos\theta)$

Resistivity:
$\rho = R\frac{A}{l}$

Conductivity:
$\sigma = \frac{1}{\rho} = \frac{l}{RA}$

E-field from vector potential:
$\vec{E} = -\nabla\phi - \frac{\partial \vec{A}}{\partial t}$

Energy of dielectric:
$W = \frac{1}{2} \int_V \vec{D} \cdot \vec{E} dV$

\subsection*{Dipoles}
Dipole moment:
$\vec{p} = q\vec{d}$

E-field from dipole:
$\vec{E} \approx \frac{-\vec{p}}{4\pi \epsilon_{\circ}r^3} \hat{r}$

Torque on dipole from E-field:
$\vec{\tau} = \vec{p} \times \vec{E}$ 

\section*{Magnetism}
Radius of circular motion in B-field:
$R = \frac{mv}{qB}$

Ampere's Law (short):
$\oint \vec{B} \cdot \vec{dl} = \mu_{\circ} I$

Volume current from Magnetization:
$\vec{J_M} = \vec{\nabla} \times \vec{M}$

Surface current from Magnetization:
$\vec{K_M} = \vec{M} \times \hat{n}$

Magnetic field of long straight wire:
$\vec{B} = \frac{\mu_{\circ} I}{2 \pi r} \hat{r}$

B-field from vector potential:
$\vec{B} = \vec{\nabla} \times \vec{A}$

Magnetization from H-field:
$M = \chi_m H$

B-field from M \& H:
$B = \mu_{\circ} (H + M) = \mu_{\circ} (1 + \chi_m)H = \mu H$

\section*{Circuits}
Induced EMF:
$emf = -N\frac{d\Phi_B}{dt}$

Kirchhoff's loop rule:
$\sum_i V_i = 0$

Charge on capacitor:
$Q = CV$

Ohm's Law:
$V=IR$

Current:
$I=\frac{dQ}{dt}$

Circuit Addition Rules
\begin{tabular}{|cc|}
\hline
Series & Parallel\\
$R_{tot} = \sum_i R_i$ & $R_{tot} = (\sum_i R_i^{-1})^{-1}$\\
$C_{tot} = (\sum_i C_i^{-1})^{-1}$ & $C_{tot} = \sum_i C_i$ \\
\hline
\end{tabular}

\section*{Optics}
Speed of Light:
$c = \frac{1}{\sqrt{\mu\epsilon}}$

Index of refraction:
$n = \frac{c}{v}$

Snell's Law:
$n_1\sin\theta_1 = n_2\sin\theta_2$

Critical angle for total internal reflection:
$n_1\sin\theta_c = n_2\sin90$

Poynting vector:
$\vec{S} = \vec{E} \times \vec{H} = \frac{1}{\mu} (\vec{E} \times \vec{B}) $

Energy density of EM-wave:
$u = \frac{1}{2} (\epsilon E^2 + \frac{1}{\mu} B^2)$

Intensity of EM-wave:
$I = <|\vec{S}|>$

Radiation pressure, incident normal:
$P = \frac{I}{c}$

Net radiation pressure:
$P_{net} = P_{incident} + P_{reflected}$

Wave equation in vacuum:
$\nabla^2\vec{E} - \mu_{\circ}\epsilon_{\circ}\ddot{\vec{E}} = 0$
$\nabla^2\vec{B} - \mu_{\circ}\epsilon_{\circ}\ddot{\vec{B}} = 0$

Wave equation in conductor:
$\nabla^2\vec{E} - \mu\sigma\dot{\vec{E}} - \mu\epsilon\ddot{\vec{E}} = 0$
$\nabla^2\vec{B} - \mu\sigma\dot{\vec{B}} - \mu\epsilon\ddot{\vec{B}} = 0$

Skin depth:
$d = \frac{1}{k}$

Skin depth ($\sigma << \epsilon\omega$):
$d \approx \sqrt{\frac{4\epsilon}{\mu\sigma^2}}$

Skin depth ($\sigma >> \epsilon\omega$):
$d \approx \sqrt{\frac{2}{\omega\mu\sigma}}$

\section*{Fundamental}
Lorentz force:
$\vec{F_L} = \vec{F_E} + \vec{F_B} = q\vec{E} + q\vec{v} \times \vec{B}$

Maxwell's Equations
\begin{tabular}{|cc|}
\hline
Integral Form & Differential Form\\
\hline
 $\oint \vec{E} \cdot \vec{dA} = \frac{Q_{enc}}{\epsilon_{\circ}}$ & $\vec{\nabla} \cdot \vec{E} = \frac{\rho}{\epsilon_{\circ}}$ \\
 $\oint \vec{B} \cdot \vec{dA} = 0$ & $\vec{\nabla} \cdot \vec{B} = 0$ \\
 $\oint \vec{E} \cdot \vec{dl} = -\frac{d}{dt} \int \vec{B} \cdot \vec{dA}$ & $\vec{\nabla} \times \vec{E} = -\frac{\partial B}{\partial t}$ \\
 $\oint \vec{B} \cdot \vec{dl} = \mu_{\circ} ( \int \vec{J} \cdot \vec{dA} + \epsilon_{\circ} \frac{d}{dt} \int \vec{E} \cdot \vec{dA} )$ & $\vec{\nabla} \times \vec{B} = \mu_{\circ} (\vec{J}  + \epsilon_{\circ} \frac{\partial E}{\partial t})$\\
\hline
\end{tabular}

\section*{Thermodynamics}

Gas Constant:
$R = N_A k_B$

Internal energy from degrees of freedom (f):
$U = \frac{f}{2} nRT$

Internal Energy of mono-atomic gas:
$U = nN_A (KE)_{avg} = nN_A \frac{3}{2}k_B T = \frac{3}{2}nRT$

Change in Internal Energy:
$\Delta U = Q - W$

Enthalpy:
$H(U,P) = U + PV$

Helmholtz Free Energy:
$F(U,S) = U - TS$

Heat absorbed:
$Q = m C \Delta T$

Ideal gas Law:
$PV = nRT = nk_B N_A T$ 

1st Law of Thermodynamics:
$dU = \dbar Q + \dbar W$

2nd Law of Thermodynamics:
$dS = \frac{\delta Q}{T}$

Work:
$W = \oint P dV$

$\dbar W = P dV$

$dH = \dbar Q + V dP$

$dU = C_v dT$

$dH = C_p dT$

$\frac{\partial \beta}{\partial T} = -\frac{1}{k_B T^2} = -\frac{\beta}{T}$

Molar Specific Heat:
$C = \frac{1}{N_{mol}} \frac{\partial U}{\partial T} = \frac{1}{N_{mol}} \frac{\partial U}{\partial \beta} \frac{\partial \beta}{\partial T}$

Efficiency of cycle:
$\eta = \frac{W_{out}}{Q_{in}} = \frac{Q_{in} - Q_{out}}{Q_{in}} = 1 - \frac{Q_{out}}{Q_{in}}$

RMS Gas Particle Speed:
$v_{rms} = \sqrt{\frac{3RT}{mN_A}}$

Quantum Volume/Concentration:
$V_q = (\frac{mk_B T}{2\pi \hbar^2})^{\frac{3}{2}}$

\section*{Micro-Canonical Ensemble}

Number of microstates in system:
$\Omega = \frac{N!}{\prod_i n_i!}$

Two-level system microstates:
$\Omega = \frac{N!}{n_1! n_2!} = \frac{N!}{n_1! (N-n_1)!}$

Entropy:
$S = k_B ln\Omega$

\section*{Canonical Ensemble}

Partition function:
$Z = \sum_i e^{-\beta E_i} = \sum_i e^{-\frac{E_i}{k_B T}}$

Energy of system:
$U = -N\frac{\partial}{\partial \beta} ln(Z)$

Average energy per particle:
$u = -\frac{\partial}{\partial \beta} ln(Z)$

\section*{Distributions}

* Note, $g_i$ is degeneracy level 

Fermi-Dirac:
$n_i = \frac{g_i}{e^{\beta (\epsilon_i - \mu} + 1}$

Bose-Einstein:
$n_i = \frac{g_i}{e^{\beta (\epsilon_i - \mu} - 1}$

Boltzmann:
$p_i = \frac{e^{-\beta \epsilon_i}}{\sum_i^M e^{-\beta \epsilon_j}}$

Maxwell-Boltzmann Speed Distribution:
$f(v) = 4\pi (\frac{mN_A}{2\pi R T})^{\frac{3}{2}} v^2 e^{\frac{-mN_Av^2}{2RT}}$

\begin{center}
     \Large{\textbf{Quantum Mechanics}} \\
     \hrule
\end{center}

Time-independent Schrodinger equation:
$H\psi = \frac{-\hbar^2}{2m} \nabla^2\psi + V\psi = E\psi$

Time-dependent Schrodinger equation:
$i\hbar \frac{\partial \psi}{\partial t} = H\psi$ 

Momentum operator:
$\hat{p} = -i\hbar \frac{\partial}{\partial x}$

\section*{Harmonic Oscillator}
$\hat{H} = \frac{\hat{p}^2}{2m} + \frac{1}{2}k\hat{x}^2 = \frac{\hat{p}^2}{2m} + \frac{1}{2}m\omega^2\hat{x}^2$

$E_n = \hbar \omega (n + \frac{1}{2})$

$a = \sqrt{\frac{m\omega}{2\hbar}}(\hat{x} + \frac{i}{m\omega}\hat{p}$

$a^{\dagger} = \sqrt{\frac{m\omega}{2\hbar}}(\hat{x} - \frac{i}{m\omega}\hat{p}$

$\hat{x} = \sqrt{\frac{\hbar}{2m\omega}}(a^{\dagger} + a)$

$\hat{p} = i\sqrt{\frac{\hbar m\omega}{2}}(a^{\dagger} - a)$

$a^{\dagger}\ket{n} = \sqrt{n+1}\ket{n+1}$

$a\ket{n} = \sqrt{n}\ket{n-1}$

$N = a^{\dagger}a \implies N\ket{n} = n\ket{n}$

$\bra{n}\hat{x}\ket{n} = 0$ same for $\hat{p}$

$\bra{n}\hat{x}\ket{n+1} = \bra{n+1}\hat{x}\ket{n} = \sqrt{\frac{\hbar(n+1)}{2m\omega}}$

$[a,a^{\dagger}]=1$

$[\hat{H},a^{\dagger}] = \hbar\omega a^{\dagger}$

\section*{Spin}
Projection of Spin:
$\vec{n} \cdot \vec{S} = n_xS_x + n_yS_y + n_zS_z$

Hamiltonian of particle in B-field:
$\hat{H}_s = -\vec{\mu} \cdot \vec{B} = -\gamma \vec{S} \cdot \vec{B}$

Dipole moment of spin particle:
$\vec{\mu} = \gamma\vec{S} = \frac{gq}{2m}\vec{S}$

$S_j = \frac{\hbar}{s} \sigma_j$

$\sigma_x = 
\begin{pmatrix}
0 & 1 \\
1 & 0
\end{pmatrix}$
$\sigma_y = 
\begin{pmatrix}
0 & -i \\
i & 0
\end{pmatrix}$
$\sigma_z = 
\begin{pmatrix}
1 & 0 \\
0 & -1
\end{pmatrix}$

$\sigma_j^2 = \hat{I}$

$\{\sigma_j,\sigma_k\} = 2\hat{I}\delta_{j,k}$

$[\hat{S}_x, \hat{S}_y] = i\hbar\hat{S}_z$

$[\hat{S}_y, \hat{S}_z] = i\hbar\hat{S}_x$

$[\hat{S}_z, \hat{S}_x] = i\hbar\hat{S}_y$

$\hat{S}^2\ket{s, m_s} = \hbar^2 s(s+1)\ket{s, m_s}$

$\hat{S}_z\ket{s, m_s} = \hbar m_s\ket{s, m_s}$

$\hat{S}_{\pm} \ket{s, m_s} = \hbar \sqrt{s(s+1)- m_s(m_s \pm 1)}\ket{s, m_s \pm 1}$

$\hat{S}_x = \frac{1}{2} (\hat{S}_+ + \hat{S}_-)$

$\hat{S}_y = \frac{1}{2i} (\hat{S}_+ - \hat{S}_-)$

$\braket{\hat{S}_x^2} = \braket{\hat{S}_y^2} = \frac{1}{2}(\braket{\hat{S}^2} - \braket{\hat{S}_z^2}) = \frac{\hbar^2}{2}[s(s+1)-m_s^2]$

\section*{Angular Momentum}
$L_z\ket{\psi} = \hbar m \ket{\psi}$

\section*{Perturbation Theory}

Perturbed Hamiltonian:
$\hat{H} = \hat{H}_0 + \hat{H}_p$

First order energy correction:
$E_n^{(1)} = \bra{n^{(0)}}\hat{H}_p\ket{n^{(0)}}$

Second order energy correction:
$E_n^{(2)} = \sum_{m\neq n } \frac{|\bra{m}\hat{H_p}\ket{n}|^2}{E_n^{(0)} - E_m^{(0)}}$

\subsection*{Time-Dependent}

Transition probability:
$P_{j\to f} = \frac{1}{\hbar^2}| \int_{t_1}^{t_2} \bra{f}\hat{V}_p\ket{j} e^{i\omega_{fj}t} dt|^2$

$\omega_{fj} = \frac{E_f - E_j}{\hbar}$

\section*{Hydrogen Atom}

Energy levels:
$E_n = -\frac{13.6 eV}{n^2} = -\frac{1 Ry}{n^2}$

\section*{General Solutions}

Square Well Wave-function:
$\psi_n (x) = \sqrt{\frac{2}{\ell}}sin(\frac{n\pi}{\ell}x)$

Square Well Energy Levels:
$E_n = \frac{n^2 \pi^2 \hbar^2}{2m\ell^2}$ 

Dirac-potential Wave-function:
$\psi_k (x) = 
\begin{cases}
Ae^{kx} & Left  \\
Be^{-kx} & Right 
\end{cases}$

Dirac-potential Energy Levels:
$E_k = -\frac{\hbar^2 k^2}{2m}$

\section*{Concepts}

$\hat{H}$ for E-field:
$\hat{H} = qE$

Expectation value or $\hat{A}$:
$\bra{\psi}\hat{A}\ket{\psi} = \int \psi^* \hat{A} \psi d^3r$

States with definite parity:
$\implies \psi^* \psi$ is symmetric

Expectation value of anti-symmetric operator with symmetric states:
$\bra{\psi}\hat{A}\ket{\psi} = 0$

\begin{center}
     \Large{\textbf{Classical Mechanics}} \\
     \hrule
\end{center}

\section*{Linear Motion}

Momentum:
$\vec{\rho} = m\vec{v}$

Kinetic energy:
$T = \frac{1}{2}mv^2$

Period of Oscillation:
$\mathcal{T} = \frac{2\pi}{\omega}$

Frequency of Oscillation:
$f = \frac{1}{\mathcal{T}} = \frac{\omega}{2\pi}$

\section*{Rotational Motion}

Kinetic energy:
$T = \frac{1}{2}I\omega^2$ 

Linear Displacement:
$s = r\theta$

Rotational velocity from linear velocity:
$\vec{\omega} = \frac{\vec{v}}{r}$

Rotational acceleration from linear acceleration:
$\vec{\alpha} = \frac{\vec{a}}{r}$

Angular Momentum from linear momentum:
$\vec{L} = \vec{r} \times \vec{\rho}$

Moment of Inertia (sum):
$I = \sum_i m_ir_i^2$

Moment of Inertia (integral):
$I = \int r^2 dm$

Parallel axis theorem:
$I = I_c + Md^2$

Perpendicular axis theorem:
$I_z = I_x + I_y$

Moment of inertia about unit vector ($\vec{u}$):
$I = \bf{u}^{\dagger} \bf{I} \bf{u}$ 

Moment of inertia tensor in xyz:
$\bf{I} = 
\begin{bmatrix}
I_x & 0 & 0 \\
0 & I_y & 0 \\
0 & 0 & I_z
\end{bmatrix}$

Angular momentum from rotational velocity:
$\vec{L} = I \vec{\omega}$

Torque:
$\vec{\tau} = \vec{r} \times \vec{F}$

\section{Lagrangian Mechanics}

Lagrangian:
$\mathcal{L} =  T - U$

Euler-Lagrange equation:
$\frac{\partial \mathcal{L}}{\partial q} - \frac{d}{dt}\frac{\partial \mathcal{L}}{\partial \dot{q}} = 0$

Canonical Momenta:
$p_i = \frac{\partial \mathcal{L}}{\partial \dot{q}}$

Hamiltonian:
$\mathcal{H} = \sum_i p_i \dot{q_i} - \mathcal{L}$

Hamilton's Equations:
$\dot{p}_i = - \frac{\partial \mathcal{H}}{\partial q_i}$ $\dot{q}_i =  \frac{\partial \mathcal{H}}{\partial p_i}$

\section*{General Solutions}

Eigenfrequency of Simple Spring:
$\omega = \sqrt{\frac{k}{m}}$

Eigenfrequency of Simple Pendulum:
$\omega = \sqrt{\frac{g}{\ell}}$

\section*{Concepts}

Conservation of Angular Momentum: No external torque is acting on the system
